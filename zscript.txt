version "4.14"

#include "zscript/CO2_Nade.zsc"


class VirCO2NadeHandler : EventHandler
{
	override void CheckReplacement(ReplaceEvent e)
	{
		switch (e.Replacee.GetClassName()) {
		
		
			case 'BlueFrag' :
			case 'HelmFrag' :
				if (!random(0, 24)) {e.Replacement = "HDCO2GrenadeAmmo";}	
				break;
				
			case 'RocketBoxRandom' :
			case 'BlurSphereReplacer' :
				if (!random(0, 12)) {e.Replacement = "HDCO2GrenadePickup";}	
				break;
			}
	}
}
